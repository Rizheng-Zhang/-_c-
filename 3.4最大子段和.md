//3.4最大子段和.h
int MaxSum(int n, int*a){
	int sum = 0, b = 0;
	for (int i = 0; i < n; i++){
		if (b > 0)//b[j-1]>0
			b += a[i];//b[j]=b[j-1]+a[j]
		else
			b = a[i];//b[j]=a[j]
		if (b > sum)
			sum = b;//不断更新sum为最大值
	}
	return sum;
}

//3.4最大子段和.cpp
#include"3.4 最大子段和.h"
#include<iostream>
using namespace std;
const int N = 1000;
int main(){
	int n,f;
	cin >> n;
	int c[N];
	for (int i = 0; i < n; i++){
		cin >> c[i];
	}
	cout<<MaxSum(n, c);//调用函数求整数数组c的最大子段和
	cin >> f;
	return 0;
}
